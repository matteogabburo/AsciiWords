```python
import ascii_word as aw

text = 'hello, world!'
s = aw.set_str(text)
aw.print_s(s, '$')
```

### Add header
```python
import ascii_word as aw

text = 'hello, world!'
s = aw.add_sequence_header(s, '>> ')
aw.print_s(s, '$')
```
